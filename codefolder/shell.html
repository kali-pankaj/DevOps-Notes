5. Arrays in Shell Scripting


    #!/bin/bash
    myarray=("apple" "banana" "cherry" 1 20 5.5)

    echo "First fruit: ${myarray[0]}"
    echo "Second fruit: ${myarray[1]}"
    echo "Third fruit: ${myarray[2]}"
    echo "All elements: ${myarray[@]}"

    echo "Number of elements: ${\#myarray[*]}" # remove space from string
    echo "Value of index 3-3: ${myarray[*]:3:3}"

    # Updating our array:
    myarray+=("new" "orange" 75000)
    echo "Updated array: ${myarray[@]}"

13. Logical Operators in Shell Scripting


    #!/bin/bash

    if [ "$age" -ge 18 ] && [ "$age" -lt 100 ]; then
        echo "You are an adult."
    else
        echo "You are not an adult."
    fi


    # Or operator (||)

    read -p "What is your country? " country
    read -p "What is your age? " age
    if [ "$country" = "INDIA" ] || [ "$age" -ge 18 ]; then
        echo "You are eligible for voting."
    else
        echo "You are not eligible."
    fi

15. Loops in Shell Scripting


    #!/bin/bash

    # For loop
    for i in {1..10}; do
        echo "Iteration: $i"
    done

    for name in "pankaj" "pradeep" "Charlie"; do
        echo "Hello, $name!"
    done

    # While loop
    count=1
    while [ "$count" -le 5 ]; do
        echo "Count: $count"
        ((count++))
    done

    # Until loop
    counter=10
    until [ "$counter" -eq 5 ]; do
        echo "Counter: $counter"
        ((counter--))
    done

    # Infinite loop
    while true; do
        echo "This will run forever. Press Ctrl+C to stop."
        sleep 1
    done


    # Infinite loop use for a specific condition
    while true; do
        read -p "Enter a number (or 'exit' to quit): " input
        if [ "$input" = "exit" ]; then
            echo "Exiting the loop."
            break
        elif [[ "$input" =~ ^[0-9]+$ ]]; then
            echo "You entered the number: $input"
        else
            echo "Invalid input. Please enter a number or 'exit'."
        fi
    done
