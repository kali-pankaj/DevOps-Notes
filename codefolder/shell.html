7. String Operations in Shell Scripting


        #!/bin/bash
        myvars="hey bro how are you ? "
        
        myvarLength=${\#myvars}
        echo "Length of the string: $myvarLength"

        echo "uppercase: ${myvars^^}"
        echo "lowercase: ${myvars,,}"
        echo "capitalize first letter: ${myvars^}"
        echo "capitalize first letter of each word: ${myvars~~}"

        To Replace a string:
        newvar=${myvars /how/where}
        echo "After replacement: $newvar"

        To slice a string:
        echo "after slicing: ${myvars:4:3}"

13. Logical Operators in Shell Scripting


        #!/bin/bash

        if [ $age -ge 18 ] && [ $age -lt 100 ]; then
            echo "You are an adult."
        else
            echo "You are not an adult."
        fi


        # Or operator (||)

        read -p "What is your country? " country
        read -p "What is your age? " age
        if [ "$country" == "INDIA" ] || [ $age -ge 18 ]; then
            echo "You are eligible for voting."
        else
            echo "You are not eligible."
        fi

15. Loops in Shell Scripting


        #!/bin/bash

        # For loop
        for i in {1..10}; do
            echo "Iteration: $i"
        done

        for name in "pankaj" "pradeep" "Charlie"; do
            echo "Hello, $name!"
        done

        # While loop
        count=1
        while [ $count -le 5 ]; do
            echo "Count: $count"
            ((count++))
        done

        # Until loop
        counter=10
        until [ $counter -eq 5 ]; do
            echo "Counter: $counter"
            ((counter--))
        done

        # Infinite loop
        while true; do
            echo "This will run forever. Press Ctrl+C to stop."
            sleep 1
        done


        # Infinite loop use for a specific condition
        while true; do
            read -p "Enter a number (or 'exit' to quit): " input
            if [ "$input" == "exit" ]; then
                echo "Exiting the loop."
                break
            elif [[ "$input" =~ ^[0-9]+$ ]]; then
                echo "You entered the number: $input"
            else
                echo "Invalid input. Please enter a number or 'exit'."
            fi
        done
